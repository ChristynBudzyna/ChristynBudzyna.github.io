Arrays

Hashes

my_hash = { name: "Christyn", city: "New York"} #creates a hash
my_hash = Hash.new #another way to create a hash
my_hash[:name] #will yield "Christyn"
my_hash[:pet] = "Dottie" #will add a key value pair to the hash

@name = name
end

end

end
